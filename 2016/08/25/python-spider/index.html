Python Spider

一直都在关注补天，发现补天上边注册了好多厂商，简单写了个爬虫小脚本爬取一下厂商列表，也方便自己查看！

1、先进行一页的爬取

#!usr/bin/env python
#-*-utf-8-*-
#author by dadongge
import requests     //导入requests库

company = []        //定义一个空列表
page = 0             //设置页值得初始值
url = "https://butian.360.cn/company/lists/page/1"    //目标url  
r = requests.get(url)        //get请求网址
r.encoding = r.apparent_encoding         //设置编码格式
html = r.text
# print html
//对获取的网页内容进行分割，获取厂商名和url部分
x = html.split('<td  align="left" height="30" style="padding-left:20px;">')[1:]
for i in x:
    name = i.split('>')
    # print name
    company.append({'name':name[1][:-3],'url':name[4][:-4]})
    # print company
for i in company:
    print i['name'],'\t',i['url']

    # print name[1][:-3],"\t",
    # print name[4][:-4]
